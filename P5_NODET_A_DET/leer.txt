#include <stdio.h>

int esPrimo(int num) {
    if (num <= 1) {
        return 0; // No es primo
    }
    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            return 0; // No es primo
        }
    }
    return 1; // Es primo
}

int main() {
    int inicio, fin;

    printf("Ingrese el inicio del rango: ");
    scanf("%d", &inicio);
    printf("Ingrese el fin del rango: ");
    scanf("%d", &fin);

    printf("NÃºmeros primos en el rango [%d, %d]:\n", inicio, fin);
    for (int i = inicio; i <= fin; i++) {
        if (esPrimo(i)) {
            printf("%d ", i);
        }
    }
    printf("\n");

    return 0;
}
